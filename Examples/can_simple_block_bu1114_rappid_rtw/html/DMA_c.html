76/*!



77 *   @brief Computes register contents and initializes DSPIx.



78 *	@par Include



79 *					DSPI.h



80 * 	@par Description



81 *					Function initializes DSPIx in specified mode (Master or Slave),



82 *					computes register configuration to achieve the desired baud rate



83 *					(if possible) and sets interruption mask. By default data are changed



84 *					on a leading edge of DSPI clock and captured on a falling edge (CPHA=1).



85 *					Inactive state of DSPI clock is set to low by default (CPOL=0). Both



86 *					parameters are changeable after the init phase using other functions



87 *					(see setPhaseSPIx and setPolaritySPIx functions for additional information).



88 * 	@param[in] DspiNumber - Number of DSPI module (0 or 1 or 2).



89 *	@param[in] mode - DSPIx mode configuration (Master or Slave).



90 *					Predefined values:



91 *					- MASTER	DSPIx will handle communication as a Master.



92 *					- SLAVE	DSPIx will listen to the incoming commands as a Slave.



93 *	@param[in] sysClk



94 *					System clock frequency [Hz].



95 *	@param[in] baudRate



96 *					Desired baud rate [b/s]. Maximal and minimal Baud rate depends on



97 *						the actual system clock frequency and on the limits of the PBR and



98 *						BR registers. Maximal and minimal values for fsys = 16 MHz are shown



99 *						in the table below, for other frequencies see the external excel file.



100 *						Predefined values:



101 *						- min baudRate = 70 b/s for fsys = 16 MHz



102 *						- max baudRate = 4 000 000 b/s for fsys = 16MHz



103 *	@param[in] intMask



104 *					32bit mask allowing us to choose which interruption we want to activate.



105 *						If no interruption is necessary, please add simply 0 instead of the whole



106 *						mask. Predefined iterruption masks can be combined arbitrarily using



107 *						logical OR (|) operator.



108 *						Predefined value:



109 *						- TCFRE - Transmission Complete Request Enable flag mask



110 *						- EOQFRE - Finished Request Enable flag mask



111 *						- TFUFRE - Transmit FIFO Underflow Request Enable flag mask



112 *						- TFFFRE - Transmit FIFO Fill Request Enable flag mask



113 *						- RFOFRE - Receive FIFO Overflow Request Enable flag mask



114 *						- RFDFRE - Receive FIFO Drain Request Enable flag mask



115 *	@remarks 	System clock frequency should be added in multiples of the basic units



116 *				(e.g. in kHz, MHz), but everytime with respect to the units of a baud rate.



117 *	@par Code sample1



118 *			DSPI_Init(0, MASTER, 16000000, 4000000, TCFRE);



119 *			- Command initializes DSPI no. 0 in Master mode with desired baud rate



120 *			4 Mbit/s (more exactly 3.81Mbit/s, because 1 Mb/s = 1 048 576 b/s) and



121 *			activated interruption when transmission is completed. System clock



122 *			frequency is 16 MHz.



123 *	@par Code sample2



124 *			DSPI_Init(2, SLAVE, 16, 4, TCFRE | EOQFRE);



125 *			- Command initializes DSPI no. 2 in Slave mode with desired baud rate



126 *			4 Mb/s and activated interruption when transmission is completed and



127 *			when request is finished.



128 ********************************************************************************/

167/******************* Prototype function for DMA_receive_init*****************************/



168void DMA_receive_Init(void)



169{



170



171#define DSPI_1_CHNL                    7                         // DMA channel 0 to be assigned



172#define DSPI_1_SOURCE                  2                         // DMA source number (SW must assign to DMA channel)



173



174  // Clear DMA_MUX for DMA channel



175  DMAMUX.CHCONFIG[DSPI_1_CHNL].R = 0x0;



176  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD0_.B.SADDR = (vuint32_t) &DSPI_0.POPR.R;// Source address



177



178  //EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD0_.B.SADDR = (vuint32_t) &TransmitBuffer[0]; // Test Source address



179  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD4_.B.SSIZE = SIZE_4_byte;// Source data transfer size



180  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD4_.B.SOFF = 4;// Source address signed offset



181  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD12_.B.SLAST = -Transmit_data_len;// Last source address adjustment



182  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD4_.B.SMOD = 0;// Source address module



183  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD16_.B.DADDR = ((vuint32_t) &ReceiveBuffer[0]);// Destination pushr DATA address



184  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD4_.B.DSIZE = SIZE_4_byte;// Destination data transfer size



185  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD20_.B.DOFF = 0;// Destination address signed offset



186  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD24_.B.DLAST_SGA = 0;// Last Distination address adjustment



187  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD4_.B.DMOD = 0;// Destination address module



188  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD8_.B.SMLOE = 0;// Source minor loop offset enable



189  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD8_.B.DMLOE = 0;// Destination minor loop offset enable



190  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD8_.B.MLOFF = 0;



191  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD8_.B.NBYTES = 4;//minor byte count     // Number of bytes to be transfered in each request



192  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD28_.B.BITER = ((vuint16_t) Receive_data_len/4);// Begining major iteration count



193  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD20_.B.CITER = ((vuint16_t) Receive_data_len/4);// Current major iteration count



194  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD20_.B.CITER_LINKCH = 0;// Link channel number



195  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD28_.B.D_REQ = 0;// disable channel when major loop is done



196  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD28_.B.INT_HALF = 0;// no interrupt when half major count complete



197  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD28_.B.INT_MAJ = 1;// set interrupt when major count complete



198  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD20_.B.CITER_E_LINK = 0;



199  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD28_.B.BITER_E_LINK = 0;



200  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD28_.B.MAJOR_E_LINK = 0;



201  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD28_.B.E_SG = 0;



202  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD28_.B.BWC = 0;



203  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD28_.B.START = 0;



204  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD28_.B.DONE = 0;



205  EDMA.CHANNEL[DSPI_1_CHNL].TCDWORD28_.B.ACTIVE = 0;



206  DMAMUX.CHCONFIG[DSPI_1_CHNL].R = 0x80 + DSPI_1_SOURCE;// ENBL + TRIG + source#1



207}

253#define DSPI_PUSHR_EOQ_BITMASK         0x08000000



254#define DSPI_PUSHR_CONT_DIS_BITMASK    0x7FFFFFFF

258void SPI_DMA_Send_Message(uint8_t *data , vuint16_t len)



259{



260  uint16_t i;



261  for (i = 0;i < len; i++) {



262    if (i == len - 1) {



263      //TransmitBuffer[i] = DSPI_PUSHR_PCS0 | DSPI_PUSHR_CTAS(0) | DSPI_PUSHR_TXDATA(data[i]);



264      TransmitBuffer[i] = DSPI_PUSHR_CS1_BITMASK | DSPI_PUSHR_CONT_EN_BITMASK |



265        DSPI_PUSHR_TXDATA(data[i]);



266    } else {



267      //TransmitBuffer[i] = DSPI_PUSHR_PCS0 | DSPI_PUSHR_CONT | DSPI_PUSHR_CTAS(0) | DSPI_PUSHR_TXDATA(data[i]);



268      TransmitBuffer[i] = DSPI_PUSHR_CS1_BITMASK | DSPI_PUSHR_CONT_EN_BITMASK |



269        DSPI_PUSHR_CTAS(0) | DSPI_PUSHR_TXDATA(data[i]);



270    }



271  }



272



273  EDMA.CHANNEL[DSPI_0_CHNL].TCDWORD28_.B.BITER = len;// Begining major iteration count



274  EDMA.CHANNEL[DSPI_0_CHNL].TCDWORD20_.B.CITER = len;// Current major iteration count



275  EDMA.CHANNEL[DSPI_0_CHNL].TCDWORD12_.B.SLAST = -len*4;// Last source address adjustment



276  for (i = 0;i < len; i++) {



277    int j = 8;



278    citer_count = EDMA.CHANNEL[DSPI_0_CHNL].TCDWORD20_.B.CITER;



279    EDMA.CHANNEL[DSPI_0_CHNL].TCDWORD28_.B.START = 1;



280



281    /* wait until minor loop complete */



282    if (EDMA.CHANNEL[DSPI_0_CHNL].TCDWORD20_.B.CITER == 1)// if it is the last minor loop



283    {



284      while (EDMA.CHANNEL[DSPI_0_CHNL].TCDWORD20_.B.CITER == 1) {



285      }



286    }



287



288    while (citer_count == EDMA.CHANNEL[DSPI_0_CHNL].TCDWORD20_.B.CITER &&



289           EDMA.CHANNEL[DSPI_0_CHNL].TCDWORD20_.B.CITER !=1) {



290    }



291



292    ;



293  }



294}

26#ifndef _LOCAL_TYPEDEFS_H_



27#define _LOCAL_TYPEDEFS_H_



28#if defined (__MWERKS__)               /* Metrowerk CodeWarrior */



29#include <stdint.h>



30



31/*  Standard typedefs used by header files, based on ISO C standard */



32typedef volatile int8_t vint8_t;



33typedef volatile uint8_t vuint8_t;



34typedef volatile int16_t vint16_t;



35typedef volatile uint16_t vuint16_t;



36typedef volatile int32_t vint32_t;



37typedef volatile uint32_t vuint32_t;



38



39#elif defined (__ghs__)                /* GreenHills */



40#include <stdint.h>



41



42/* Standard typedefs used by header files, based on ISO C standard */



43typedef volatile int8_t vint8_t;



44typedef volatile uint8_t vuint8_t;



45typedef volatile int16_t vint16_t;



46typedef volatile uint16_t vuint16_t;



47typedef volatile int32_t vint32_t;



48typedef volatile uint32_t vuint32_t;



49



50#else



51



52/* This is needed for compilers that don't have a stdint.h file */



53typedef signed char int8_t;



54typedef unsigned char uint8_t;



55typedef volatile signed char vint8_t;



56typedef volatile unsigned char vuint8_t;



57typedef signed short int16_t;



58typedef unsigned short uint16_t;



59typedef volatile signed short vint16_t;



60typedef volatile unsigned short vuint16_t;



61typedef signed long int32_t;



62typedef unsigned long uint32_t;



63typedef volatile signed long vint32_t;



64typedef volatile unsigned long vuint32_t;



65



66#endif

91#ifndef tBool



92



93typedef unsigned char tBool;           /*!< basic boolean type */



94



95#ifndef FALSE



96#define FALSE                          ((tBool)0)                /*!< Boolean type FALSE constant */



97#endif



98



99#ifndef TRUE



100#define TRUE                           ((tBool)1)                /*!< Boolean type TRUE constant */



101#endif



102



103typedef unsigned char tU8;             /*!< unsigned 8-bit integer type */



104typedef signed char tS8;               /*!< signed 8-bit integer type */



105typedef unsigned short tU16;           /*!< unsigned 16-bit integer type */



106typedef signed short tS16;             /*!< signed 16-bit integer type */



107typedef unsigned long tU32;            /*!< unsigned 32-bit integer type */



108typedef signed long tS32;              /*!< signed 32-bit integer type */



109typedef signed long long tS64;         /*!< signed 64-bit integer type */



110typedef unsigned long long tU64;       /*!< unsigned 64-bit integer type */



111typedef tS16 tFrac16;                  /*!< 16-bit signed fractional Q1.15 type */



112typedef tS32 tFrac32;                  /*!< 32-bit Q1.31 type */



113typedef float tFloat;                  /*!< single precision float type */



114typedef double tDouble;                /*!< double precision float type */

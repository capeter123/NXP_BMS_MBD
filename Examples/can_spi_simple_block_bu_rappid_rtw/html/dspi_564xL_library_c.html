38  /******************************************************************************



39   *   Function:  dspi_564xl_read_master



40     @brief    Reads the data from DSPI module  - master.



41     @details   Reads the data from DSPI module



42



43     @param[in]     dspi_num     DSPI number



44



45     @return Data from DSPI module



46   */



47  uint16_t dspi_564xl_read_master(uint8_t dspi_num)



48  {



49    uint16_t ret = 0U;



50    while (DSPI[dspi_num]->SR.B.RFDF != 1) {



51    }                                  /* Wait for Receive FIFO Drain Flag = 1 */



52



53    ret = (uint16_t)DSPI[dspi_num]->POPR.R;/* Read data received by master SPI */



54    DSPI[dspi_num]->SR.R = 0x00020000; /* Clear RDRF flags*/



55    return ret;



56  }

60     @brief    Reads the data from DSPI module - Slave .



61     @details   Reads the data from DSPI module

64



65     @return Data from DSPI module



66   */



67  uint16_t dspi_564xl_read_slave(uint8_t dspi_num)



68  {



69    uint16_t ret = 0U;

72



73    ret = (uint16_t)DSPI[dspi_num]->POPR.R;/* Read data received by slave SPI */



74    DSPI[dspi_num]->SR.R = 0x00020000; /* Clear RDRF flags*/

78  /******************************************************************************



79   *   Function:  dspi_564xl_send



80     @brief    Send the data from DSPI module



81     @details   Send  the data from DSPI module



82



83     @param[in]     dspi_num     DSPI number



84     @param[in]     pushr_data     data



85



86     @return none



87



88   */



89  void dspi_564xl_send(uint8_t dspi_num, DSPI_PUSHR_32B_tag pushr_data)



90  {



91    DSPI[dspi_num]->SR.R = DSPI_ISR_TCF_RE | DSPI_ISR_EOQF_RE;/* TCF EOQF */



92    DSPI[dspi_num]->PUSHR.R = pushr_data.R;



93    return;



94  }

96  /******************************************************************************



97   *   Function:  dspi_564xl_config



98     @brief    Configure DSPI module .



99     @details   Configure DSPI module



100



101     @param[in]     dspi_num     DSPI number



102     @param[in]     dspi_MCR              DSPI Module Configuration Register settings



103     @param[in]     dspi_CTAR             DSPI Clock and Transfer Attributes Register



104



105     @return none



106   */



107  void dspi_564xl_config(uint8_t dspi_num, uint32_t dspi_MCR, uint32_t dspi_CTAR)



108  {



109    uint32_t mcr = dspi_MCR;



110    uint32_t ctar = dspi_CTAR;



111    DSPI[dspi_num]->MCR.R = mcr;



112    DSPI[dspi_num]->CTAR[0].R = ctar;



113    DSPI[dspi_num]->MCR.B.HALT = 0x0;



114    return;



115  }

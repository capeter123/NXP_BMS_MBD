95 *



96 * Inputs:          uint8_t *data - pointer to data

104static void SPI_Send_Message(uint8_t *data , uint8_t len)



105{



106  DSPI_PUSHR_32B_tag pushr;



107  uint8_t i;



108  pushr.R = 0U;



109



110  // Keep chip select enabled



111  for (i = 0; i < len-1; i++) {



112    pushr.B.TXDATA = data[i];



113    pushr.R |= DSPI_PUSHR_CS6_BITMASK;



114    pushr.R |= DSPI_PUSHR_CONT_EN_BITMASK;



115    DSPI[0]->PUSHR.R = pushr.R;



116  }



117



118  // Disable chip select after last data byte



119  pushr.B.TXDATA = data[i];



120  pushr.R &= DSPI_PUSHR_CONT_DIS_BITMASK;



121  DSPI[0]->PUSHR.R = pushr.R;



122



123  // Indicate end of queue



124  DSPI[0]->SR.R = DSPI_ISR_TCF_RE | DSPI_ISR_EOQF_RE;



125}

420/******************************************************************************



421 * Function name:   Command_LTC



422 *



423 * Inputs:



424 *



425 * Outputs:         16 bit PEC



426 *



427 * Description:     Calculate PEC based on input data



428 *****************************************************************************/



429static void Command_LTC(uint8_t tx_cmd[2])



430{



431  uint8_t cmd[4];



432  uint16_t cmd_pec;



433  cmd[0] = tx_cmd[0];



434  cmd[1] = tx_cmd[1];



435  cmd_pec = Calculate_PEC15(cmd, CMD_LEN);



436  cmd[2] = (uint8_t)(cmd_pec >> 8);



437  cmd[3] = (uint8_t)(cmd_pec);



438  SPI_Send_Message(cmd, 4);



439}

441/******************************************************************************



442 * Function name:   Calculate_PEC15



443 *



444 * Inputs:          uint8_t *data - pointer to data



445 *                  uint8_t len - number of data bytes



446 *



447 * Outputs:         16 bit PEC



448 *



449 * Description:     Calculate PEC based on input data



450 *****************************************************************************/



451static uint16_t Calculate_PEC15(uint8_t *data, uint8_t len)



452{



453  uint16_t rem, addr;



454  uint8_t i;



455



456  // PEC Seed



457  rem = 16;



458  for (i = 0; i < len; i++) {



459    addr = ((rem >> 7) ^ data[i]) & 0xFF;//calculate PEC table address



460    rem = (rem << 8 ) ^ PEC15Table[addr];



461    rem &= 0xFFFF;



462  }



463



464  return ((rem << 1) & 0xFFFF);



465}
